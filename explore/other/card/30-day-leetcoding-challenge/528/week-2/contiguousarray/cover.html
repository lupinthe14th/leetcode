package contiguousarray

func findMaxLength(nums []int) int {
        // まず　0と1の数を数える
        count := make(map[int]int)
        for _, n := range nums {
                count[n]++
        }
        // 最大部分配列を把握する
        l := min(count[0], count[1]) * 2

        // 最大部分配列から、インデックスをずらして、条件に合致するか確認。
        // 合致しなければ一つずづ短くして行く
        for l >= 0 {
                for i := 0; i < len(nums)-l+1; i++ {
                        if isContiguousArray(nums[i : l+i]) {
                                return l
                        }
                }
                l -= 2
        }
        return l
}

func min(x, y int) int {
        if x < y {
                return x
        }
        return y
}

func isContiguousArray(nums []int) bool {
        c := make(map[int]int)
        for _, n := range nums {
                c[n]++
        }
        if c[0] == c[1] {
                return true
        }
        return false
}
